}
async function doScreenshot(el, filename) {
  el.classList.add('screenshot-mode');
  const tableWraps = el.querySelectorAll('.table-wrap');
  const saved = [];
  tableWraps.forEach(tw => { saved.push({el:tw, ox:tw.style.overflowX, w:tw.style.width}); tw.style.overflowX='visible'; tw.style.width='max-content'; });
  const hasCharts = el.querySelectorAll('canvas').length > 0;

  await new Promise(r => setTimeout(r, 60));
  try {
