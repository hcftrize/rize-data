  const canvas=document.getElementById('exp-vp-chart');if(!canvas)return;

  const now=Date.now()/1000;

  // Build historical VP correctly:
  // At bond creation time T, maturity = 0 (timeMarker = T, so age = 0)
  // As time passes, maturity grows linearly until fullMatS
  // For each bond, use its createdAtTimestamp as the timeMarker origin
  // (before any increaseBond recalculation)
  const allEvents=[...allBondCreated.map(e=>({ts:parseInt(e.timestamp),amount:parseAmt(e.amount),nftId:e.nftId}))].sort((a,b)=>a.ts-b.ts);

    runRize+=ev.amount;
    const d=tsToDate(ev.ts.toString());
    // At creation time: maturity = 0, boost = 1x — this is correct
    // DO NOT use current time to compute historical maturity
    // The bond just got created: timeMarker = ev.ts, so (ev.ts - ev.ts)/fullMatS = 0
